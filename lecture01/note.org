  # fibonacci

  if [ $# -eq 0 ]; then
      ehco "needd parameters"
      exit 1
  fi

  fib() {
      N = "$1"

      if [ -z "${N##*[!0-9]*}"]; then
          echo "negative"
          exit 1
      fi
      
  }

  fib "$1"
